/** Returns false if at least one of the curr/delq boxes is checkmarked **/
	var isBlankForm = function (element_checkbox) {
		var blankForm = false;
		var selected =false;
		$('.checkBoxes').each( function(i) {
		     if (this.checked){
		        selected = true;
		        return false;
		    }
		});		
		if (!selected) {
			$('.checkBoxes1').each( function(i) {
		    	 if (this.checked){

/** Initializes the checkboxes in their default state. Attempts to 'pre-select' all boxes that have the 
    same PID, sequentially, until one is different

	var initializeForm = function(alsoCheckMarkIt) {
		var checkboxes = [];
		var checkboxes1 = [];
		var pids = [];
		var pidsC = [];
		var pidsD = [];


		try{
			checkboxes = document.getElementsByClassName("checkBoxes");		// delinquent checkboxes
			checkboxes1 = document.getElementsByClassName("checkBoxes1");	// current checkboxes
			pids = document.getElementsByClassName("pids");
			pidsC = document.getElementsByClassName("pidsC");
			pidsD = document.getElementsByClassName("pidsD");
		}catch(err){
			//solve issue with IE not supporting "getElementByClassName"
			var delinqRows = document.getElementById("etaxTemplateForm:payments_").children[1].children.length;
			var pz = 0;
			for(var i = 0; i < delinqRows; i++){
				checkboxes[i] = document.getElementById("etaxTemplateForm:payments_:"+i+":delinquentCB");
				pidsD[i] = document.getElementById("etaxTemplateForm:payments_:"+i+":delinquentPID");
				pids[pz++] = document.getElementById("etaxTemplateForm:payments_:"+i+":delinquentPID");
			}

			var currRows = document.getElementById("etaxTemplateForm:payments1").children[1].children.length;
			for(var y = 0; i < currRows; i++){
				checkboxes1[i] = document.getElementById("etaxTemplateForm:payments1:"+i+":paymentCB");
				pidsC[i] = document.getElementById("etaxTemplateForm:payments_:"+i+":currentPID");
				pids[pz++] = document.getElementById("etaxTemplateForm:payments_:"+i+":currentPID");
			}
		};

		// get first unique pid
		var pid = '';
		for (var i=0; i < pids.length; i++){		
			if (pids[i] != null && pids[i].value != '' && pids[i].value != null){
				pid = pids[i].value;

	/**
	*	IE ONLY: Prevent double click by determining the time of clicks and
	*	evaluating them on an allowed threshold.*/
	function preventDoubleClick(){
		if(navigator.userAgent.match(/MSIE/)!==null){
			if((Date.now() - clickEventTime) < clickEventTimeThreshold){
				return true;
			}
			clickEventTime = Date.now();
		}
		return false;
	}

	/**
	* Verify selected checkbox is allowed to be checked.
	*/
	var onSelectValidate = function(obj) {
	
		/** identify the first selected PID; only items with that 
		  * same pid can be selected
		  **/
	  var selectedPid = getFirstPID(obj);
	  if (selectedPid == null || selectedPid == ''){
	  	  return true;
	  }
	  if (!obj.checked){
	  	/** un-checking is always allowed **/
	  	return true;
	  }
	  
	  var elementId = (obj).id;
	  var elementIdTokens = (obj).id.split(':');
	  
	  var thisElementIndex = Number(elementIdTokens[2]);
	  var refName = 'currentPID';
	  if (elementIdTokens[3] == "paymentCB"){
	  	refName = 'currentPID';
	  } else {
	  	refName = 'delinquentPID';
	  }
	  
	  /** find the PID value of the item being selected; then validate if it's allowed  **/
	  var pidElementId = elementIdTokens[0] +"\\:"+ elementIdTokens[1] +"\\:"+ elementIdTokens[2] +"\\:"+ refName;
	  var pidObjs = $('#' + pidElementId);
	  if (pidObjs.length > 0){
		  var pidObj = pidObjs[0];
		  var myPid = pidObj.value;
		
		  if (selectedPid != '' && selectedPid != null && selectedPid != myPid){
		  	  /** Element is not allowed to pay, since it does not match existing PID.**/
		  	  displayMultiPIDNotice();
		  	  obj.checked = false;
		  	  obj.disabled = true;
		  	  return false;
		  }
	  }else {
			;;
			///debug('ERROR: cannot find PID element for the clicked object.');	  
	  }
	  
	  return true;
	};

	    try{
	      var prevElement = document.getElementById(prevElementId);

	        && prevElement.checked == false
			    && element_checkbox.checked == true){
	        alert("Error: Delinquent taxes must be paid in reverse chronological order, without skipping any years.");
	        element_checkbox.checked = false;
	        element_checkbox.disabled = true;
	      }
	    } catch (err){
	      ;
	    }

	  }
	};

	/**
	* Calculate total payment amount over the rendered tables rows.
	* For each table row, add the applied amount to a total. If the 
	* row contains the 'select' checkbox, verify it is checked. 
	*/
	var calculateTotalPaymentByRows = function(){
	  var currentTableRows = new Array();
	  var delinquentTableRows = new Array();

	  /** Get amount rows */
	  var currentTable = document.getElementById('etaxTemplateForm:payments1');
	  if(currentTable != null){
	  	currentTableRows = (currentTable).getElementsByTagName('TBODY')[0].children;
	  }

	  var delinquentTable = document.getElementById('etaxTemplateForm:payments_');
	  if(delinquentTable != null){
	  	delinquentTableRows = (delinquentTable).getElementsByTagName('TBODY')[0].children;
	  }


	  /**
	  * Calculate total payment by iterating each row in order to get each APPLIED
	  * amount.
	  *   */
	  var totalCurrentYearSelected = Number(0);
	  var totalDelinquentYearSelected = Number(0);

	  for(var int_i=0; int_i<currentTableRows.length; int_i++){
	    var elementAmount = document.getElementById('etaxTemplateForm:payments1:'+int_i+':appliedAmount1');
	    var elementAmountValue = (elementAmount==null) ? Number(0) : Number(elementAmount.innerHTML.replace("$","").replace(",",""));
	        elementAmountValue = isNaN(elementAmountValue) ? Number(0) : elementAmountValue;

		/* look for checkbox */
		var rowCheckbox = document.getElementById('etaxTemplateForm:payments1:'+int_i+':paymentCB');
		if(rowCheckbox != null && !rowCheckbox.checked) {
			elementAmountValue = Number(0);
		}

	    totalCurrentYearSelected = Number(totalCurrentYearSelected) + Number(elementAmountValue);
	  }

	  for(var int_i=0; int_i<delinquentTableRows.length; int_i++){
